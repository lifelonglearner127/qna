Create a Wizard

POST: http://host.com/api/api.php
{
    "endpoint": "wizards",
    "action": "create",
    "params": {"name": "Something new"}
}
                                    
{
    "status":true,
    "wizard_id":3
}

Get a Wizard

POST: http://host.com/api/api.php
{
    "endpoint": "wizards",
    "action": "get",
    "params": {"id": 3}
}
                                    
{
    "status":true,
    "wizard":{
        "id":"2",
        "name":"something"
    }
}

Delete a Wizard

POST: http://host.com/api/api.php
{
    "endpoint": "wizards",
    "action": "delete",
    "params": {"id": 3}
}
                                    
{
    "status":true
}

Getting all Subjects

POST: http://host.com/api/api.php
{
    "endpoint": "subjects",
    "action": "get_all",
    "params": {"wizard_id":2}
}
                                    
{
    "status":true,
    "subjects":[
        {
            "id":"2",
            "question":"Is that correct?",
            "wizard_id":"2",
            "type_id":"1",
            "value":"{\"caption\":\"title\",\"value\":\"\",\"weight\":100}"
        }
    ]
}

Create a Subject

POST: http://host.com/api/api.php
{
    "endpoint": "subjects",
    "action": "create",
    "params": {
        "wizard_id":2,
        "type_id":1,
        "question":"Who are you?"
    }
}
                                    
{
    "status":true,
    "subject_id":3
}

Get a Subject

POST: http://host.com/api/api.php
{
    "endpoint": "subjects",
    "action": "get",
    "params": {"id": 3}
}
                                    
{
    "status":true,
    "subject":{
        "id":"2",
        "question":"Is that correct?",
        "wizard_id":"2",
        "type_id":"1",
        "value":"{\"caption\":\"title\",\"value\":\"\",\"weight\":100}"
    }
}

Update a Subject

POST: http://host.com/api/api.php
{
    "endpoint": "subjects",
    "action": "update",
    "params": {
        "id":2,
        "wizard_id":2,
        "type_id":1,
        "question":"Is that true?"
    }
}
                                    
{
    "status":true
}
